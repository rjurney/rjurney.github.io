Using groupby and PrettyTable in Python

# What I really want to see is... un-reciprocal links. Lots of to/not a lot of to back & vice versa
to_from_ratios = []
for key, group in groupby(relative_links, key=lambda x: (x[0], x[2])):
    for key2, group2 in groupby(relative_links, key=lambda x: (x[2], x[0])):
        if key == key2:
            to_from_ratios.append((key[0], key[1], list(group)[0][3]/list(group2)[0][3]))

sorted_to_from_ratios = sorted(to_from_ratios, key=lambda x: x[2] * -1)
print Table(["From", "To", "Ratio"],sorted_to_from_ratios)
